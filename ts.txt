        private void button1_Click_1(object sender, EventArgs e)
        {
            using (var wc = new WebClient())
            {
                //Подключение к localhost
                var body = wc.DownloadString("http://localhost:4444/TransferSimulator/fullName ");

                //Получаем данные страницы
                var js = Newtonsoft.Json.Linq.JObject.Parse(body);

                //Переводим кодировку, чтобы текст выводился читаемым
                byte[] utf = Encoding.GetEncoding("WINDOWS-1251").GetBytes(js.ToString());
                string a = Encoding.UTF8.GetString(utf);

                //Убираем {"value": " "}, чтобы не мешало тесту
                a = a.Replace("{", "");
                a = a.Replace("}", "");
                a = a.Replace(":", "");
                a = a.Replace("value", "");
                a = a.Replace(@"""", "");

                //Присваеваем значение законченной переменной label
                label1.Text = a;
            }
        }

        private void button2_Click_1(object sender, EventArgs e)
        {
            string результатПроверки;
            string статусТеста;

            // Проверка на запрещенные символы
            if (label1.Text.Intersect(@"@#№$;%:^?&*()/|\+=-").Any())
            {
                результатПроверки = "Найдены запрещённые символы";
                статусТеста = "Не успешно";
            }
            else
            {
                результатПроверки = "Нет запрещённых символов";
                статусТеста = "Успешно";
            }

            // Отображаем в label2 (для информации)
            label2.Text = результатПроверки;

            try
            {
                string path = Application.StartupPath + "//ТестКейс.docx";

                Word.Application word = new Word.Application();
                Word.Document doc = word.Documents.Open(path);

                // Получаем таблицу
                Word.Table table = doc.Tables[1];

                // Проверяем, существует ли строка с номером currentRow
                if (currentRow > table.Rows.Count)
                {
                    table.Rows.Add();
                }

                // Заполняем ячейки корректными данными
                table.Cell(currentRow, 1).Range.Text = label1.Text;  // Что проверяли
                table.Cell(currentRow, 2).Range.Text = результатПроверки;  // Ожидаемый результат
                table.Cell(currentRow, 3).Range.Text = статусТеста;  // Реальный результат

                currentRow++;

                doc.Save();
                doc.Close();
                word.Quit();
                MessageBox.Show($"Готово! Заполнена строка {currentRow - 1}");
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка: " + ex.Message);
            }
        }
